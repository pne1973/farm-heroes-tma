let farm = { eggs: 0, virtualCash: 10.00, // Moeda para a Quinta (ex: BioCoins) realBalance: 0.00, // Saldo para Levantamento (€) // Taxas de Conversão (Exemplo) virtualRate: 0.50, // Ganha 0.50 BC por cada ovo realRate: 0.0001 // Ganha 0.0001€ por cada ovo (CPM 1€) }; // FUNÇÃO ÚNICA DE VENDA function sellEggs() { if (farm.eggs >= 1) { let eggsToSell = Math.floor(farm.eggs); // 1. Calcula o ganho virtual para reinvestir let gainVirtual = eggsToSell * farm.virtualRate; farm.virtualCash += gainVirtual; // 2. Calcula o ganho real para a carteira let gainReal = eggsToSell * farm.realRate; farm.realBalance += gainReal; // Reseta os ovos farm.eggs = 0; updateUI(); // Notificação de sucesso com os dois valores alert(`Venda Concluída!\n+ ${gainVirtual.toFixed(2)} BC (Quinta)\n+ ${gainReal.toFixed(4)} € (Real)`); } }
